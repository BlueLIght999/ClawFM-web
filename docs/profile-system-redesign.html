核心挑战：项目当前仅提取了 NeteaseCloudMusicApi 返回的 6 个基本字段（id, title, artist, album, durationMs, coverUrl），但 API 实际提供了大量未使用的接口：artist/detail（歌手详情）、artist/desc（歌手简介）、style/list（曲风标签体系）、song/wiki/summary（歌曲百科）、simi/song（相似歌曲）、simi/artist（相似歌手）、playmode/intelligence/list（智能播放推荐）等。

解决方案（四级降级策略）：采集层采用 网易云 API 优先 → 网络搜索 → LLM 兜底 → 标记未知 的四级策略获取歌曲/歌手元数据。第一级直接调用网易云已有的 10+ 个未使用接口获取结构化数据（免费），命中率达 70-80%。搜索不到时再使用 MusicBrainz/百科搜索，最后才使用 LLM 推断。同时将网易云的推荐与搜索接口与画像系统深度集成，增强歌单管理与 DJ 推荐能力。

// ─── 第一级：网易云 API（免费，结构化数据，命中率 70-80%） ───
async function enrichFromNeteaseAPI(songId, title, artist) {
  // 1. song/detail — 获取 entertainmentTags、awardTags（可能含风格标签）
  const detail = await musicSource.songDetail(songId);

  // 2. song/wiki/summary — 歌曲百科（可能含流派、年代、背景）
  const wiki = await musicSource.songWiki(songId);

  // 3. artist/detail + artist/desc — 歌手详情和简介
  const artistDetail = await musicSource.artistDetail(artistId);
  const artistDesc = await musicSource.artistDesc(artistId);

  // 4. style/list — 获取全曲风标签体系，构建反向映射
  //    用 style_song 按风格获取歌曲，判断歌曲属于哪些风格
  const styleTags = await musicSource.styleList();
  // → 构建风格→歌曲映射，用于反向推断歌曲风格

  // 解析并提取结构化元数据
  const metadata = parseNeteaseMetadata(detail, wiki, artistDetail, artistDesc);
  if (metadata) return { ...metadata, _source: 'netease_api', _confidence: 0.92 };

  return null; // 第一级未命中
}

// ─── 第二级：网络搜索（免费，半结构化，命中率 15-20%） ───
async function enrichFromWebSearch(title, artist) {
  const mbResult = await searchMusicBrainz(title, artist);
  if (mbResult) return { ...mbResult, _source: 'musicbrainz', _confidence: 0.88 };

  const wikiResult = await searchWiki(`${artist} ${title} 歌曲 风格`);
  if (wikiResult) return { ...wikiResult, _source: 'wiki', _confidence: 0.82 };

  const neteaseTag = await searchNeteaseTags(title, artist);
  if (neteaseTag) return { ...neteaseTag, _source: 'netease_tag', _confidence: 0.85 };

  return null; // 第二级未命中
}

// ─── 第三级：LLM 兜底（低成本，推断，命中率 5-10%） ───
async function enrichFromLLM(songs) {
  // 固定前缀（被缓存）+ 可变后缀
  const result = await callLLM(SONG_LLM_PROMPT_PREFIX, songs);
  return result.map(r => ({ ...r, _source: 'llm', _confidence: 0.70 }));
}

// ─── 第四级：标记未知（不参与推荐决策） ───
function markUnknown(songId) {
  return { songId, _source: 'unknown', _confidence: 0,
           genre: null, mood: null, _note: '所有来源均未命中' };
}

// 主流程：依次尝试四级
async function enrichSongMetadata(song) {
  return (await enrichFromNeteaseAPI(song.id, song.title, song.artist))
      || (await enrichFromWebSearch(song.title, song.artist))
      || (await enrichFromLLM([song]))?.[0]
      || markUnknown(song.id);
}

// server/domain/profile/analyzers/RecommendationEnhancer.js
class RecommendationEnhancer extends BaseAnalyzer {
  async analyze(profile) {
    // 1. 从画像提取推荐参数
    const params = this._buildParams(profile);

    // 2. 并行调用多个推荐 API（Promise.all 降低延迟）
    const [styleSongs, simiArtists, intelligence, personalized] = await Promise.all([
      this._fetchByStyleTags(params.topGenres),        // style_song
      this._fetchBySimiArtists(params.topArtists),     // simi_artist → artist_songs
      this._fetchIntelligence(params.currentSongId),   // playmode_intelligence_list
      this._fetchPersonalized(params.mood),             // personalized
    ]);

    // 3. 合并 + 去重 + 按画像权重排序
    const candidates = this._mergeAndRank(
      [...styleSongs, ...simiArtists, ...intelligence, ...personalized],
      profile
    );

    // 4. 输出增强推荐参数供 recommender.js 使用
    return {
      genreHints: params.topGenres,           // 回填到 fillQueue 的 genreHints
      styleTagIds: params.styleTagIds,        // style_song 的 tagId
      moodKeywords: params.moodKeywords,      // cloudsearch 的情绪关键词
      similarArtistIds: params.similarArtistIds,
      intelligenceSeedId: params.currentSongId,
      rankedCandidates: candidates.slice(0, 30),
    };
  }

  _buildParams(profile) {
    return {
      topGenres: profile.tags.MusicPreference.genre
        .filter(t => t.weight > 0.3)
        .slice(0, 3)
        .map(t => t.value),
      topArtists: profile.tags.ArtistPreference.topArtists.slice(0, 5),
      mood: profile.tags.EmotionProfile.currentMood?.value || '平静',
      currentSongId: this.queueStore.current?.id,
    };
  }

TagMapper — 标签映射器

将 RawDataChunk 中的每条记录映射到标签体系中的标签。例如，一首 genre: "pop" 的歌曲映射为 MusicPreference.genre = "pop" 的证据项。

WeightCalculator — 权重计算器

// 权重计算公式
weight = frequencyScore * recencyScore * confidenceScore

// frequencyScore: 基于播放频次的归一化分数
//   = min(playCount / maxPlayCount, 1.0)
//   例: 该流派播放 50 次 / 最多的流派播放 80 次 = 0.625

// recencyScore: 基于最近播放时间的衰减分数
//   = decayFactor ^ (weeksSinceLastPlay)
//   例: 2 周前最后播放, decayFactor=0.95 → 0.95^2 = 0.9025

// confidenceScore: 基于数据量 + LLM 置信度
//   = min(evidenceCount / 10, 1.0) * avgLLMConfidence
//   例: 8 条证据 / 10 = 0.8 * LLM平均置信度 0.85 = 0.68

ConflictResolver — 冲突解决器

当同一维度出现冲突标签时（如 LLM 对同一歌曲推断出不同流派），按置信度取高者；若置信度接近，保留两者并降低各自权重。

DecayApplier — 衰减应用器

每次画像更新时，对所有已有标签应用时间衰减。超过 4 周未被证据支撑的标签，权重降至 0.1 以下时自动标记为"休眠"。

3.4 权重计算与衰减机制

因子计算方式说明
频次权重playCount / maxPlayCount该标签关联歌曲的播放总次数，归一化到 0-1
时间衰减0.95 ^ weeksSinceLastPlayEbbinghaus 遗忘曲线，每周衰减 5%
置信度min(evidenceCount/10, 1) * avgLLMConfidence证据越多越可信，LLM 置信度加权
跳过惩罚weight *= (1 - skipRate * 0.5)高跳过率降低该标签权重
完播奖励weight *= (1 + completeRate * 0.2)高完播率提升该标签权重

4.4 用户聚类分析模块 (UserClusterAnalyzer) — 无监督学习

特征工程

将 UserProfile 中的标签转化为数值特征向量，供聚类算法消费：

// server/domain/profile/analyzers/UserClusterAnalyzer.js

/**
 * 将 UserProfile 标签转化为特征向量
 * 每个维度归一化到 0-1，使用标签的 weight 值
 */
function buildFeatureVector(profile) {
  const features = {};

  // 1. 音乐流派特征（10 维）
  const genres = ['pop','rock','folk','electronic','classical','jazz','hip-hop','rnb','metal','country'];
  const genreTags = profile.tags.MusicPreference?.genre || [];
  genres.forEach(g => {
    const tag = genreTags.find(t => t.value === g);
    features[`genre_${g}`] = tag ? tag.weight : 0;
  });

  // 2. 情绪特征（8 维）
  const moods = ['欢快','忧伤','平静','激昂','怀旧','浪漫','治愈','暗黑'];
  const moodTags = profile.tags.MusicPreference?.mood || [];
  moods.forEach(m => {
    const tag = moodTags.find(t => t.value === m);
    features[`mood_${m}`] = tag ? tag.weight : 0;
  });

  // 3. 歌手地区特征（5 维）
  const regions = ['华语','欧美','日韩','东南亚','其他'];
  const regionTags = profile.tags.ArtistPreference?.regions || [];
  regions.forEach(r => {
    const tag = regionTags.find(t => t.value === r);
    features[`region_${r}`] = tag ? tag.weight : 0;
  });

  // 4. 行为习惯特征（6 维）
  features['habit_activePeriod_morning'] = profile.tags.BehaviorHabit?.activePeriod?.value === '早晨型' ? 1 : 0;
  features['habit_activePeriod_afternoon'] = profile.tags.BehaviorHabit?.activePeriod?.value === '下午型' ? 1 : 0;
  features['habit_activePeriod_evening'] = profile.tags.BehaviorHabit?.activePeriod?.value === '晚间型' ? 1 : 0;
  features['habit_activePeriod_night'] = profile.tags.BehaviorHabit?.activePeriod?.value === '夜猫型' ? 1 : 0;
  features['habit_skipRate'] = parseSkipRate(profile.tags.BehaviorHabit?.skipRate?.value);
  features['habit_listenIntensity'] = parseIntensity(profile.tags.BehaviorHabit?.listenIntensity?.value);

  // 5. 聊天风格特征（4 维）
  features['chat_commandFreq'] = profile.tags.ChatStyle?.commandFrequency || 0;
  features['chat_avgLength'] = normalizeLength(profile.tags.ChatStyle?.avgMessageLength);
  features['chat_instructional'] = profile.tags.ChatStyle?.interactionMode === '指令型' ? 1 : 0;
  features['chat_demanding'] = profile.tags.ChatStyle?.satisfactionPattern === '高要求' ? 1 : 0;

  // 总维度: 10 + 8 + 5 + 6 + 4 = 33 维
  return features;
}

聚类算法

算法适用场景选择理由
K-Means（主要） 单用户电台场景：将历史快照作为多个"虚拟用户"进行聚类 实现简单、可解释性强、簇中心即为典型画像

单用户场景的聚类策略

当前系统为单用户电台，聚类策略是将用户的历史画像快照作为多个时间节点的"虚拟用户"，发现用户自身的偏好演变模式：

class UserClusterAnalyzer extends BaseAnalyzer {
  async analyze(profile) {
    // 1. 获取历史画像快照（最近 30 个）
    const snapshots = await profileSnapshotManager.recent(30);

    // 2. 为每个快照构建特征向量
    const featureVectors = snapshots.map(s => buildFeatureVector(s.profile));

    // 3. 加入当前画像
    featureVectors.push(buildFeatureVector(profile));

    // 4. 自动确定最佳 K 值（轮廓系数 Silhouette Score）
    const optimalK = this._findOptimalK(featureVectors);

    // 5. 执行 K-Means 聚类
    const clusters = this._kmeans(featureVectors, optimalK);

    // 6. 为每个簇生成可解释标签
    const labeledClusters = clusters.map(cluster => ({
      clusterId: cluster.id,
      centroid: cluster.centroid,
      label: this._generateClusterLabel(cluster.centroid),  // 自动标签
      memberSnapshots: cluster.members,
      size: cluster.members.length,
    }));

    // 7. 确定当前画像所属簇
    const currentCluster = labeledClusters[labeledClusters.length - 1];

    return {
      optimalK,
      clusters: labeledClusters,
      currentClusterId: currentCluster.clusterId,
      currentClusterLabel: currentCluster.label,
      clusterDistribution: this._computeDistribution(labeledClusters),
    };
  }

  /**
   * 根据簇中心特征自动生成标签
   * 取权重最高的 3 个特征维度组合为标签
   */
  _generateClusterLabel(centroid) {
    const sorted = Object.entries(centroid)
      .sort(([,a], [,b]) => b - a)
      .slice(0, 3);
    const parts = sorted.map(([k, v]) => {
      // genre_pop → "流行", mood_怀旧 → "怀旧"
      const [, value] = k.split('_');
      return value;
    });
    return parts.join('·');  // 例: "pop·怀旧·晚间"
  }

  /**
   * 轮廓系数自动选 K（2-8 范围）
   */
  _findOptimalK(vectors) {
    let bestK = 2, bestScore = -1;
    for (let k = 2; k <= Math.min(8, vectors.length - 1); k++) {
      const result = this._kmeans(vectors, k);
      const score = this._silhouette(vectors, result);
      if (score > bestScore) { bestScore = score; bestK = k; }
    }
    return bestK;
  }
}

// ─── 默认实现：频次×时间衰减×置信度 ───
class DefaultWeightStrategy extends WeightStrategy {
  get name() { return 'default'; }
  calculate(evidence, context) {
    const freqScore = Math.min(evidence.length / 10, 1.0);
    const recencyScore = Math.pow(0.95, this._weeksSince(evidence[0]?.playedAt));
    const confScore = evidence.reduce((s, e) => s + (e.confidence || 0.7), 0) / evidence.length;
    const skipPenalty = context.skipRate ? (1 - context.skipRate * 0.5) : 1;
    return freqScore * recencyScore * confScore * skipPenalty;
  }
}

// server/domain/profile/builders/SchemaMigrator.js
class SchemaMigrator {
  constructor() {
    this.migrations = new Map();  // version → migrate function
  }

  register(fromVersion, toVersion, migrateFn) {
    this.migrations.set(`${fromVersion}→${toVersion}`, migrateFn);
  }

  migrate(profile) {
    let current = profile.schemaVersion;
    while (current < TARGET_SCHEMA_VERSION) {
      const key = `${current}→${current + 1}`;
      const migrateFn = this.migrations.get(key);
      if (!migrateFn) throw new Error(`No migration for ${key}`);
      profile = migrateFn(profile);
      profile.schemaVersion = ++current;
    }
    return profile;
  }
}
